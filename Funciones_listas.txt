lista[:]

print(lista[:6])
# Devuelve desde el primer hasta el último de dos en dos

print(lista[-3:])

print(lista[::-1])

    #Crear una lista con los 5 primeros números:
        lista = [i for i in range(1,6)]

    #Crear una lista con los cuadrados de los cinco primeros números:
        lista = [i ** 2 for i in range(1,6)]

    #Crear una lista con los números pares de los 10 primeros números:
        lista = [i for i in range(1,11) if i % 2 == 0]
